95  <script>
  
96    $(document).ready(function() {
  
97      var $list = $('.signature-list');
  
98      var $checkboxes = $list.find('input[name="id"]');
  
 99      var $selectAll = $list.find('input[name="select_all"]');
  
100
  
101      var selectedIds = function() {
  
102        var ids = [];
  
103
  
104        $checkboxes.filter(':checked').each(function() {
  
105          ids.push($(this).val());
  
106        });
  
107
  
108        return ids.join();
  
109      };
  
110
  
111      var actionDisabled = function(action) {
  
112        var $checked = $checkboxes.filter(':checked');
  
113        var disabled = false;
  
114
  
115        if ($checked.length == 0) {
  
116          return true;
  
117        }
  
118
  
119        $checked.each(function() {
  
120          var $tr = $(this).closest('tr');
  
121          var $action = $tr.find('input[type=submit].' + action + '-action');
  
122
  
123          if ($action.length == 0) {
  
124            disabled = true;
  
125          }
  
126        });
  
127
  
128        return disabled;
  
129      };
  
130
  
131      var enableDisableActions = function() {
  
132        $('.validate-action input[type=submit]').prop('disabled', actionDisabled('validate'));
  
133        $('.invalidate-action input[type=submit]').prop('disabled', actionDisabled('invalidate'));
  
134        $('.subscribe-action input[type=submit]').prop('disabled', actionDisabled('subscribe'));
  
135        $('.unsubscribe-action input[type=submit]').prop('disabled', actionDisabled('unsubscribe'));
  
136        $('.delete-action input[type=submit]').prop('disabled', actionDisabled('delete'));
  
137      };
  
138
  
139      $selectAll.on('click', function(e) {
  
140        var checked = false;
  
141
  
142        if ($(this).prop('checked')) {
  
143          checked = true;
  
144        }
  
145
  
146        $checkboxes.each(function() {
  
147          if ($(this).prop('disabled') == false) {
  
148            $(this).prop('checked', checked);
  
149          }
  
150        });
  
151
  
152        enableDisableActions();
  
153      });
  
154
  
155      $checkboxes.on('click', function(e) {
  
156        enableDisableActions();
  
157      });
  
158
  
159      $('.inline-actions form').on('submit', function() {
  
160        $(this).find('input[name="selected_ids"]').val(selectedIds());
  
161      });
  
162    });
  
163  </script>
